- [ ] 千分位
```js
function ThousandFormat(num){
  // return num.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",")
  if(Math.abs(num)<1000){
    return num.toString()
  }
  let sign = num < 0? '-' : ''
  num = Math.abs(num)
  let arr = num.toString().split('')
  for(let i = arr.length-3; i>0; i-=3){
    arr.splice(i,0,',')
  }
  return sign + arr.join('')
}

```
